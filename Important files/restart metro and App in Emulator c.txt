# reset-build.ps1
# Use this script to clean and rebuild NUKKAD React Native project

Write-Host "🧹 Cleaning Node modules and caches..."
rm -r -Force .\node_modules
rm -Force package-lock.json
npm cache clean --force

Write-Host "📦 Installing dependencies..."
npm install

Write-Host "⚙️ Cleaning Android build..."
cd android
.\gradlew clean
cd ..

Write-Host "🧹 Resetting Metro Bundler cache..."
npx react-native start --reset-cache --minify &

Write-Host "📱 Rebuilding the app for Android..."
npx react-native run-android

Write-Host "✅ Done! App should launch on emulator/device."
